package arrayandarraylist;

public class ArrayMultidimensionali {
public static void main(String[] args) {
String[][] classi = new String[3][7];

classi[0][0] = "Ari";
classi[0][1] = "Gigi";
classi[0][2] = "Mike";

classi[1][0] = "Jack";
classi[1][1] = "Gaetano";
classi[1][2] = "Miri";
classi[1][3] = "Alessandro";

classi[2][0] = "Rob";
classi[2][1] = "Peppe";
classi[2][2] = "Riccardo";

System.out.println("Primo alunno della terza classe: " + classi[2][0]);

for(int i = 0; i < classi.length; i++){
for(int j = 0; j < classi[i].length && classi[i][j] != null; j++){
System.out.println("Studente " + j + " della classe" + i + ": " + classi[i][j]);
}
}
}
}
